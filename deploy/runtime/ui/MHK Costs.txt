equations{ '$MIMO$ mhk modeled cost calculations' } = define() {
	obj = ssc_create();
   
	ssc_var(obj,'device_rated_power', ${device_rated_power});
	ssc_var(obj,'system_capacity', ${system_capacity});
	ssc_var(obj,'devices_per_row', ${devices_per_row});
	
	//ssc_var(obj, 'device_type', 0);
	//ssc_var(obj,'marine_energy_tech', 0);
	
	ssc_var(obj, 'inter_array_cable_length', ${inter_array_cable_length});
	ssc_var(obj,'riser_cable_length', ${riser_cable_length});
	ssc_var(obj, 'export_cable_length', ${export_cable_length});  
	
	tech = technology();
	if (tech == 'MHKwave')
	{
		if (${wave_device_name} == 'RM3') ssc_var (obj, 'device_type', 1);
		else if (${wave_device_name} == 'RM5') ssc_var (obj, 'device_type', 2);
		else if (${wave_device_name} == 'RM6') ssc_var(obj, 'device_type', 3);
	}
	else ssc_var(obj, 'device_type', 0); //generic device and all tidal devices
		
	if (tech == 'MHKwave') ssc_var( obj, 'marine_energy_tech', 0 );
	else ssc_var( obj, 'marine_energy_tech', 1); //tidal

	success = ssc_exec(obj, 'mhk_costs');
	
	${structural_assembly_cost_modeled} = ssc_var( obj, 'structural_assembly_cost_modeled' );
	${power_takeoff_system_cost_modeled} = ssc_var( obj, 'power_takeoff_system_cost_modeled' );
	${mooring_found_substruc_cost_modeled} = ssc_var( obj, 'mooring_found_substruc_cost_modeled' );

	${development_cost_modeled} = ssc_var( obj, 'development_cost_modeled' );
	${eng_and_mgmt_cost_modeled} = ssc_var( obj, 'eng_and_mgmt_cost_modeled' );
	${plant_commissioning_cost_modeled} = ssc_var( obj, 'plant_commissioning_cost_modeled' );
	${site_access_port_staging_cost_modeled} = ssc_var( obj, 'site_access_port_staging_cost_modeled' );
	${assembly_and_install_cost_modeled} = ssc_var( obj, 'assembly_and_install_cost_modeled' );
	${other_infrastructure_cost_modeled} = ssc_var( obj, 'other_infrastructure_cost_modeled' );

	${array_cable_system_cost_modeled} = ssc_var( obj, 'array_cable_system_cost_modeled' );
	${export_cable_system_cost_modeled} = ssc_var( obj, 'export_cable_system_cost_modeled' );
	${onshore_substation_cost_modeled} = ssc_var( obj, 'onshore_substation_cost_modeled' );
	${offshore_substation_cost_modeled} = ssc_var( obj, 'offshore_substation_cost_modeled' );
	${other_elec_infra_cost_modeled} = ssc_var( obj, 'other_elec_infra_cost_modeled' );

	ssc_free(obj);
};
