on_change{'btnSaveMHKWaveResourceData'} = define() {
	logmsg('btnSaveMHKWaveResourceData');
	file = choose_file( '', 'Save MHK Wave Resource Data As', 
		'Data Files (*.csv)|*.csv', true);
		
	if ( file != '' ) {
		file = replace(file, '\\', '/');
		if (group_write('MHK Wave Resource',file))
		{	
			value( 'mhk_wave_resource_data_filename', file );
			msgbox("Data saved! \n");
		}
		else
		{
			value( 'mhk_wave_resource_data_filename', "" );
			msgbox("Data save failed! \n");
		}
	}
};

on_change{'btnLoadMHKWaveResourceData'} = define() {
	logmsg('running btnLoadMHKWaveResourceData');
	file = choose_file( '', 'Open MHK Wave Resource Data File', 
		'Data Files (*.csv)|*.csv');
		
	if ( file != '' ) {
		file = replace(file, '\\', '/');
		if (group_read('MHK Wave Resource',file))
		{	
			value( 'mhk_wave_resource_data_filename', file );
			msgbox("Data loaded! \n");
		}
		else
		{
			value( 'mhk_wave_resource_data_filename', "" );
			msgbox("Data load failed! \n");
		}
	}
};
