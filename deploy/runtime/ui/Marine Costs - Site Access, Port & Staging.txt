equations {'staging_port_cost_total'} = define() {
	return ${staging_port_cost}+${quay_cost}+${laydown_area_cost}+
		${assembly_areas_cost}+${personnel_facilities_cost}+
		${fabrication_facilities_cost};
};

equations {'port_fees_cost_total'} = define() {
	return ${port_fees_cost} + ${entrance_exit_fees_cost}+${quayside_docking_fees_cost}+
		${wharfage_fees_cost}+${other_site_access_cost};
};

equations {'site_access_port_staging_cost_total'} = define() {
	return ${staging_port_cost_total}+${cranage_cost}+${port_improvement_cost}+
		${port_fees_cost_total}+${other_site_access_cost};
};

equations {'site_access_port_staging_cost'} = define() {
	return ?${site_access_port_staging_cost_method}
			[
				${site_access_port_staging_cost_input}*${system_capacity},
				${site_access_port_staging_cost_input},
				${site_access_port_staging_cost_modeled},
				${site_access_port_staging_cost_total}
			];
};

equations {'itemized_check_site_access'} = define() {
	return ?${site_access_port_staging_cost_method}
			[
				0,
				0,
				2,
				1
			];
};

function enable_itemized_site_access()
{
	itemized_value = value('itemized_check_site_access');
	enable('staging_port_cost',itemized_value==1);
	enable('staging_port_cost_total',itemized_value==1);
	enable('quay_cost',itemized_value==1);
	enable('laydown_area_cost',itemized_value==1);
	enable('assembly_areas_cost',itemized_value==1);
	enable('personnel_facilities_cost',itemized_value==1);
	enable('fabrication_facilities_cost',itemized_value==1);
	enable('other_staging_port_cost',itemized_value==1);
	enable('cranage_cost',itemized_value==1);
	enable('port_improvement_cost',itemized_value==1);
	enable('port_fees_cost',itemized_value==1);
	enable('port_fees_cost_total',itemized_value==1);
	enable('entrance_exit_fees_cost',itemized_value==1);
	enable('quayside_docking_fees_cost',itemized_value==1);
	enable('wharfage_fees_cost',itemized_value==1);
	enable('other_port_fees_cost',itemized_value==1);
	enable('other_site_access_cost',itemized_value==1);
	enable('eng_and_mgmt_cost_input',itemized_value!=1);
	show('site_access_port_staging_cost_total',itemized_value==1);
	show('site_access_port_staging_cost_total_label',itemized_value==1);
	show('site_access_port_staging_cost_input',(itemized_value==0 || itemized_value==2));
	show('site_access_port_staging_cost_input_label',(itemized_value==0 || itemized_value==2));
	/*if (value('itemized_check_eng')==1)
	{
		enable_itemized_level_eng();
	}*/
	refresh();
}
