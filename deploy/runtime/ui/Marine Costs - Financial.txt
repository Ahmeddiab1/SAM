on_load{'Marine Costs - Financial'} = define() {
	on_change{'itemized_check_contingency'};
	enable_contingency();
	on_change{'itemized_check_construction_insurance'};
	enable_construction_insurance();
	on_change{'itemized_check_reserve_accounts'};
	enable_reserve_accounts();
	on_change{'itemized_check_other_financial'};
	enable_other_financial();
	on_change{'financial_cost_override'};
	override_financial_cost();
};

on_change{'itemized_check_contingency'} = define() {
	enable_contingency();
};

function enable_contingency() 
{
	itemized_value = value('itemized_check_contingency');
	enable('project_contingency_cost_input',itemized_value!=1);
	
	refresh();
}

on_change{'itemized_check_construction_insurance'} = define() {
	enable_construction_insurance();
};

function enable_construction_insurance()
{
	itemized_value_const = value('itemized_check_construction_insurance');
	enable('insurance_during_construction_cost_input',itemized_value_const!=1);
	
	refresh();
}

on_change{'itemized_check_reserve_accounts'} = define() {
	enable_reserve_accounts();
};

function enable_reserve_accounts()
{
	itemized_value_reserve = value('itemized_check_reserve_accounts');
	enable('maintenance_reserve_account_cost',itemized_value_reserve==1);
	enable('detailed_cost_choice_reserve_accounts',itemized_value_reserve==1);
	enable('debt_service_reserve_account_cost',itemized_value_reserve==1);
	enable('decommissioning_reserve_account_cost',itemized_value_reserve==1);
	enable('other_reserve_account_cost',itemized_value_reserve==1);
	enable('reserve_accounts_cost_input',itemized_value_reserve!=2);
	show('reserve_accounts_cost_total',itemized_value_reserve==1);
	show('reserve_total_label',itemized_value_reserve==1);
	show('reserve_accounts_cost_input',(itemized_value_reserve==0 || itemized_value_reserve==2));
	show('reserve_input_label',(itemized_value_reserve==0 || itemized_value_reserve==2));
	
	
	refresh();
}

on_change{'itemized_check_other_financial'} = define() {
	enable_other_financial();
};

function enable_other_financial() 
{
	itemized_value_other = value('itemized_check_other_financial');
	enable('other_financial_cost_input',itemized_value_other!=1);
	
	refresh();
}

on_change{'financial_cost_override'} = define() {
	override_financial_cost();
};

function override_financial_cost()
{
	override_value = value('financial_cost_override');
	enable('financial_cost_total',override_value!=1);
	enable('financial_cost_input',override_value==1);
	show('financial_cost_input',override_value==1);
	show('financial_cost_total',override_value!=1);
	
	refresh();
}
