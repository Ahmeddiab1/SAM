on_load{'Marine Costs - Financial'} = define() {
	on_change{'financial_cost_override'};
	override_financial_cost();
};

on_change{'financial_cost_override'} = define() {
	override_financial_cost();
};

function override_financial_cost()
{
	override_value = value('financial_cost_override');
	enable('financial_cost_total',override_value!=1);
	enable('financial_cost_input',override_value==1);
	show('financial_cost_input',override_value==1);
	show('financial_cost_total',override_value!=1);
	
	//Project Contingency Budget Inputs
	enable('project_contingency_budget_cost',override_value==0);
	//enable('project_contingency_cost_input',override_value==0);
	enable('project_contingency_cost_method',override_value==0);
	//enable('project_contingency_cost_modeled',override_value==0);
	
	//Insurance during Construction Inputs
	enable('insurance_during_construction_cost',override_value==0);
	//enable('insurance_during_construction_cost_input',override_value==0);
	enable('insurance_during_construction_cost_method',override_value==0);
	//enable('insurance_during_construction_cost_modeled',override_value==0);
	
	//Reserve Accounts Inputs
	enable('reserve_accounts_cost',override_value==0);
	//enable('reserve_accounts_cost_input',override_value==0);
	enable('reserve_accounts_cost_method',override_value==0);
	//enable('reserve_accounts_cost_modeled',override_value==0);
	//enable('reserve_accounts_cost_total',override_value==0);
	
	//Other Financial Cost Inputs
	enable('other_financial_cost',override_value==0);
	//enable('other_financial_cost_input',override_value==0);
	enable('other_financial_cost_method',override_value==0);
	//enable('other_financial_cost_modeled',override_value==0);
	
	refresh();
}
