	{

on_load{'Marine Costs - Plant Commissioning'} = define() {
	on_change{'itemized_check_commissioning'};
	//on_change{'level_eng'};
	enable_itemized_commissioning();
	//enable_itemized_level();
	on_change{'override_plant'};
	check_override_commissioning();
};

on_change{'override_plant'} = define() {
	check_override_commissioning();
};


function check_override_commissioning()
{
	check_value = value('override_plant');
	enable('plant_commissioning_cost_method',check_value!=1);
	enable('plant_commissioning_cost',check_value!=1);
	show('test_label',check_value==1);
	show('plant_override_label',check_value==1);
	
	refresh();
}

on_change{'itemized_check_commissioning'} = define() {
	enable_itemized_commissioning();
};

function enable_itemized_commissioning()
{
	itemized_value = value('itemized_check_commissioning');
	enable('plant_commissioning_cost_input',itemized_value!=1);
	
}
