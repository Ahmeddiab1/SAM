{
	if (value('itemized_check_development')==1)
	{
		detailed_value = value('detailed_cost_choice_financing_incentive');
		enable('financing_and_incentives_cost',detailed_value==0);
		show('financing_and_incentives_cost',detailed_value==0);
		enable('financing_and_incentives_cost_total',detailed_value);

		enable('due_diligence_cost',detailed_value);
