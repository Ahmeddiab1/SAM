		return ${outfitting_structure_cost_total};
	}

on_load{'Marine Costs - Mooring'} = define() {
	on_change{'itemized_check_mooring'};
	enable_itemized_mooring();
	on_change{'detailed_cost_choice_outfitting_structure'};
	on_change{'detailed_cost_choice_ancillary_marine_systems'};
	detailed_cost_outfitting_structure();
	detailed_cost_ancillary_marine_systems();
};

on_change{'detailed_cost_choice_outfitting_structure'} = define() {
	detailed_cost_outfitting_structure();
};

function detailed_cost_outfitting_structure()
{
	if (value('itemized_check_mooring')==1)
	{
		detailed_value = value('detailed_cost_choice_outfitting_structure');
		enable('outfitting_structure_cost',detailed_value==0);
		show('outfitting_structure_cost',detailed_value==0);
		enable('outfitting_structure_cost_total',detailed_value);
		show('outfitting_structure_cost_total',detailed_value);
		enable('vessel_landing_cost',detailed_value);
		enable('service_platforms_and_decks_cost',detailed_value);
		enable('ladders_cost',detailed_value);
		enable('railings_cost',detailed_value);
		enable('marine_coatings_cost',detailed_value);
		enable('other_outfitting_structure_cost',detailed_value);
		
		refresh();
	}
}

on_change{'detailed_cost_choice_ancillary_marine_systems'} = define() {
	detailed_cost_ancillary_marine_systems();
};

function detailed_cost_ancillary_marine_systems()
{
	if (value('itemized_check_mooring')==1)
	{
		detailed_value = value('detailed_cost_choice_ancillary_marine_systems');
		enable('ancillary_marine_systems_cost',detailed_value==0);
		show('ancillary_marine_systems_cost',detailed_value==0);
		enable('ancillary_marine_systems_cost_total',detailed_value);
		show('ancillary_marine_systems_cost_total',detailed_value);
		enable('cathodic_protection_system_cost',detailed_value);
		enable('personnel_access_system_mooring_cost',detailed_value);
		enable('ballast_system_mooring_cost',detailed_value);
		enable('condition_monitoring_cost',detailed_value);
		enable('other_ancillary_marine_system_cost',detailed_value);
		
		refresh();
	}
}

on_change{'itemized_check_mooring'} = define() {
	enable_itemized_mooring();
};
	

function enable_itemized_mooring()
{
	itemized_value = value('itemized_check_mooring');
	enable('detailed_cost_choice_outfitting_structure',itemized_value==1);
	enable('detailed_cost_choice_ancillary_marine_systems',itemized_value==1);
	enable('mooring_lines_cost',itemized_value==1);
	enable('anchors_and_piles_cost',itemized_value==1);
	enable('connecting_hardware_cost',itemized_value==1);
	enable('subfoundation_or_bedding_cost',itemized_value==1);
	enable('messenger_lines_and_buoys_cost',itemized_value==1);
	enable('scour_protection_cost',itemized_value==1);
	enable('outfitting_structure_cost',itemized_value==1);
	enable('outfitting_structure_cost_total',itemized_value==1);
	enable('vessel_landing_cost',itemized_value==1);
	enable('service_platforms_and_decks_cost',itemized_value==1);
	enable('ladders_cost',itemized_value==1);
	enable('railings_cost',itemized_value==1);
	enable('marine_coatings_cost',itemized_value==1);
	enable('other_outfitting_structure_cost',itemized_value==1);
	enable('ancillary_marine_systems_cost',itemized_value==1);
	enable('ancillary_marine_systems_cost_total',itemized_value==1);
	enable('cathodic_protection_system_cost',itemized_value==1);
	enable('personnel_access_system_mooring_cost',itemized_value==1);
	enable('ballast_system_mooring_cost',itemized_value==1);
	enable('condition_monitoring_cost',itemized_value==1);
	enable('other_ancillary_marine_system_cost',itemized_value==1);
	enable('integration_assembly_testing_checkout_cost',itemized_value==1);
	enable('transportation_cost',itemized_value==1);
	enable('other_mooring_cost',itemized_value==1);
	enable('mooring_found_substruc_cost_input',itemized_value!=2);
	show('mooring_found_substruc_cost_total',itemized_value==1);
	show('mooring_cost_label',itemized_value==1);
	show('mooring_found_substruc_cost_input',(itemized_value==0 || itemized_value==2));
	show('mooring_input_label',(itemized_value==0 || itemized_value==2));
	
	if (itemized_value==1)
	{
		detailed_cost_outfitting_structure();
		detailed_cost_ancillary_marine_systems();
	}
	
	refresh();
}
