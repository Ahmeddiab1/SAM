on_load{'Marine Costs - Mooring'} = define() {

function enable_itemized_mooring()
{
	itemized_value = value('itemized_check_mooring');
	enable('mooring_lines_cost',itemized_value==1);
	enable('anchors_and_piles_cost',itemized_value==1);
	enable('connecting_hardware_cost',itemized_value==1);
	enable('subfoundation_or_bedding_cost',itemized_value==1);
	enable('messenger_lines_and_buoys_cost',itemized_value==1);
	enable('scour_protection_cost',itemized_value==1);
	enable('outfitting_structure_cost',itemized_value==1);
	enable('vessel_landing_cost',itemized_value==1);
	enable('service_platforms_and_decks_cost',itemized_value==1);
	enable('ladders_cost',itemized_value==1);
	enable('railings_cost',itemized_value==1);
	enable('marine_coatings_cost',itemized_value==1);
	enable('ancillary_marine_systems_cost',itemized_value==1);
	enable('cathodic_protection_system_cost',itemized_value==1);
	enable('personnel_access_system_mooring_cost',itemized_value==1);
	enable('ballast_system_mooring_cost',itemized_value==1);
	enable('condition_monitoring_cost',itemized_value==1);
	enable('integration_assembly_testing_checkout_cost',itemized_value==1);
	enable('transportation_cost',itemized_value==1);
	enable('other_mooring_cost',itemized_value==1);
	enable('mooring_found_substruc_cost_input',itemized_value!=2);
	show('mooring_cost_total',itemized_value==1);
	show('mooring_cost_label',itemized_value==1);
	show('mooring_found_substruc_cost_input',(itemized_value==0 || itemized_value==2));
	show('mooring_input_label',(itemized_value==0 || itemized_value==2));
	
	refresh();
}
