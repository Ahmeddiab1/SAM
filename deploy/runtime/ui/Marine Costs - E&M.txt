equations {'construction_management_cost_total'} = define() {
	return ${salaries_eng_cost}+
		${sales_general_and_administrative_eng_cost}+${profit_eng_cost}+
		${other_construction_management_cost};
};

equations {'construction_management_cost_total_check'} = define() {
	if (${detailed_cost_choice_construction_management}==0)
	{
		return ${construction_management_cost};
	}
	else
	{
		return ${construction_management_cost_total};
	}
};

equations {'health_safety_and_environmental_monitoring_cost_total'} = define() {
	return ${health_and_safety_monitoring_cost}+
		${environmental_monitoring_cost}+${other_health_and_safety_cost};
};

		return ${health_safety_and_environmental_monitoring_cost_total};

equations {'eng_and_mgmt_cost'} = define() {
	return ?${eng_and_mgmt_cost_method}

				${eng_and_mgmt_cost_input},
				${eng_and_mgmt_cost_modeled},
				${eng_and_mgmt_cost_total}
			];
};

on_load{'Marine Costs - E&M'} = define() {
	on_change{'itemized_check_eng'};
	//on_change{'level_eng'};
	enable_itemized_eng();
	//enable_itemized_level();
	on_change{'detailed_cost_choice_construction_management'};
	on_change{'detailed_cost_choice_health_safety'};
	detailed_cost_construction_management();
	detailed_cost_health_safety();
};

on_change{'detailed_cost_choice_construction_management'} = define() {
	detailed_cost_construction_management();
};

function detailed_cost_construction_management()
{
	if (value('itemized_check_eng')==1)
	{
		detailed_value = value('detailed_cost_choice_construction_management');
		enable('construction_management_cost',detailed_value==0);
		show('construction_management_cost',detailed_value==0);
		enable('construction_management_cost_total',detailed_value);
		show('construction_management_cost_total',detailed_value);
		enable('salaries_eng_cost',detailed_value);
		enable('sales_general_and_administrative_eng_cost',detailed_value);
		enable('profit_eng_cost',detailed_value);
		enable('other_construction_management_cost',detailed_value);
		
		refresh();
	}
}

on_change{'detailed_cost_choice_health_safety'} = define() {
	detailed_cost_health_safety();
};

function detailed_cost_health_safety()
{
	if (value('itemized_check_eng')==1)
	{
		detailed_value = value('detailed_cost_choice_health_safety');
		enable('health_safety_and_environmental_monitoring_cost',detailed_value==0);
		show('health_safety_and_environmental_monitoring_cost',detailed_value==0);
		enable('health_safety_and_environmental_monitoring_cost_total',detailed_value);
		show('health_safety_and_environmental_monitoring_cost_total',detailed_value);
		enable('health_and_safety_monitoring_cost',detailed_value);
		enable('environmental_monitoring_cost',detailed_value);
		enable('other_health_and_safety_cost',detailed_value);
		
		refresh();
	}
}

function enable_itemized_eng()
{
	itemized_value = value('itemized_check_eng');

	enable('sales_general_and_administrative_eng_cost',itemized_value==1);
	enable('salaries_eng_cost',itemized_value==1);
	enable('profit_eng_cost',itemized_value==1);
	enable('project_certification_cost',itemized_value==1);
	enable('health_safety_and_environmental_monitoring_cost',itemized_value==1);
	enable('health_and_safety_monitoring_cost',itemized_value==1);
	enable('environmental_monitoring_cost',itemized_value==1);
	enable('other_eng_cost',itemized_value==1);
	enable('other_construction_management_cost',itemized_value==1);
	enable('other_health_and_safety_cost',itemized_value==1);
	show('level_eng',itemized_value==1);
	show('level_label',itemized_value==1);
	enable('eng_and_mgmt_cost_input',itemized_value==0);
	show('eng_and_mgmt_cost_total',itemized_value==1);

	show('level_line',itemized_value==1);
	/*if (value('itemized_check_eng')==1)
	{
		enable_itemized_level_eng();

}
