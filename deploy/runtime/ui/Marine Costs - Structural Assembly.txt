function enable_itemized_structural()
{
	itemized_value = value('itemized_check_structural');
	enable('prime_mover_cost',itemized_value==1);
	enable('primary_struct_assem_cost',itemized_value==1);
	enable('buoyancy_chamber_cost',itemized_value==1);
	enable('external_ballast_chambers_cost',itemized_value==1);
	enable('active_structure_cost',itemized_value==1);
	enable('additional_structural_components_cost',itemized_value==1);
	enable('marine_systems_cost',itemized_value==1);
	enable('personnel_access_system_cost',itemized_value==1);
	enable('ballast_system_cost',itemized_value==1);
	enable('navigation_lighting_cost',itemized_value==1);
	enable('coatings_cost',itemized_value==1);
	enable('transport_structure_cost',itemized_value==1);
	enable('other_costs',itemized_value==1);
}
